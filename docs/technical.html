// background.js or dedicated analytics module
const GA_MEASUREMENT_ID = 'G-XXXXXXXXXX';

function initGA() {
  const script = document.createElement('script');
  script.src = `https://www.googletagmanager.com/gtag/js?id=${GA_MEASUREMENT_ID}`;
  script.onload = () => {
    window.dataLayer = window.dataLayer || [];
    function gtag(){dataLayer.push(arguments);}
    gtag('js', new Date());
    gtag('config', GA_MEASUREMENT_ID, {
      send_page_view: false // Manual pageview tracking
    });
  };
  document.head.appendChild(script);
}

// Track custom events
function trackEvent(eventName, params) {
  if (window.gtag) {
    window.gtag('event', eventName, params);
  }
}

A/B Testing in Extensions

Test different monetization strategies, pricing, and UI elements to optimize conversion rates. Use feature flags combined with analytics.

Client-Side A/B Testing

For simple UI tests, use client-side assignment with storage synchronization across extension contexts.

ab-testing.js - Simple Implementation

const EXPERIMENTS = {
  pricing_page_v2: {
    variants: ['control', 'variant_a', 'variant_b'],
    weights: [0.5, 0.25, 0.25]
  },
  cta_button_color: {
    variants: ['blue', 'green'],
    weights: [0.5, 0.5]
  }
};

async function getVariant(experimentId) {
  // Check existing assignment
  const stored = await chrome.storage.local.get([`exp_${experimentId}`]);
  if (stored[`exp_${experimentId}`]) {
    return stored[`exp_${experimentId}`];
  }
  
  // Assign new variant using weighted random
  const exp = EXPERIMENTS[experimentId];
  const rand = Math.random();
  let cumulative = 0;
  let variant = exp.variants[0];
  
  for (let i = 0; i < exp.variants.length; i++) {
    cumulative += exp.weights[i];
    if (rand < cumulative) {
      variant = exp.variants[i];
      break;
    }
  }
  
  // Persist assignment
  await chrome.storage.local.set({
    [`exp_${experimentId}`]: variant
  });
  
  // Track assignment event
  trackEvent('experiment_assigned', {
    experiment_id: experimentId,
    variant: variant
  });
  
  return variant;
}

async function fetchExperimentConfig() {

    console.error('Experiment config error:', error);
    return null;
  }
}

Proper update management ensures your users have the latest monetization features while maintaining stability. Chrome provides built-in auto-update, but you need strategies for controlled rollouts.

// Listen for extension updates
chrome.runtime.onUpdateAvailable.addListener((details) => {
  console.log(`Update available: ${details.version}`);
  
  // Track update availability
  trackEvent('extension_update_available', {
    current_version: chrome.runtime.getManifest().version,
    new_version: details.version

// Check for updates manually
function checkForUpdates() {
  chrome.runtime.requestUpdateCheck((status, details) => {
    if (status === 'update_available') {
      console.log(`Update found: ${details.version}`);
    } else if (status === 'no_update') {
      console.log('No update available');
    }
  });
}

Staged Rollouts

Control the rate of update distribution to catch issues before affecting all users.

staged-rollout.js - Gradual Distribution

const STAGED_ROLLOUT = {
  enabled: true,
  percentage: 10, // Start with 10%
  min_version: "1.0.0"
};

async function shouldAutoUpdate() {
  const currentVersion = chrome.runtime.getManifest().version;
  
  // Check if version meets minimum requirement
  if (compareVersions(currentVersion, STAGED_ROLLOUT.min_version) < 0) {
    return true; // Always update critical versions
  }
  
  if (!STAGED_ROLLOUT.enabled) {
    return false; // Staging disabled
  }
  
  // Deterministic user assignment based on ID
  const userId = await getUserId();
  const hash = simpleHash(userId + currentVersion);
  const bucket = hash % 100;
  
  return bucket < STAGED_ROLLOUT.percentage;
}

function simpleHash(str) {
  let hash = 0;
  for (let i = 0; i < str.length; i++) {
    hash = ((hash << 5) - hash) + str.charCodeAt(i);
    hash |= 0;
  }
  return Math.abs(hash);
}

Performance Monitoring

Slow extensions hurt conversion rates. Monitor performance metrics to ensure your monetization features don't degrade user experience.

Core Web Vitals for Extensions

Track paint timings and interaction latency in your extension's popups and pages.

performance.js - Core Web Vitals

// Track popup performance
function measurePopupPerformance() {
  const metrics = {
    fcp: null,
    lcp: null,
    fid: null,
    cls: null
  };
  
  // First Contentful Paint
  const fcpEntry = performance.getEntriesByType('paint')
    .find(e => e.name === 'first-contentful-paint');
  if (fcpEntry) metrics.fcp = fcpEntry.startTime;
  
  // Largest Contentful Paint
  const lcpEntry = performance.getEntriesByType('largest-contentful-paint')
    .pop();
  if (lcpEntry) metrics.lcp = lcpEntry.startTime;
  
  // Cumulative Layout Shift
  const clsEntries = performance.getEntriesByType('layout-shift')
    .filter(e => !e.hadRecentInput);
  metrics.cls = clsEntries.reduce((sum, e) => sum + e.value, 0);
  
  // Track metrics
  trackEvent('popup_performance', {
    fcp: Math.round(metrics.fcp),
    lcp: Math.round(metrics.lcp),
    cls: metrics.cls.toFixed(3)
  });
}

// Measure when popup loads
document.addEventListener('DOMContentLoaded', measurePopupPerformance);

API Response Time Tracking

Monitor latency for your license validation and payment APIs.

api-monitoring.js - Track API Latency

async function trackedFetch(url, options = {}) {
  const startTime = performance.now();
  const endpoint = new URL(url).pathname;
  
  try {
    const response = await fetch(url, options);
    const duration = performance.now() - startTime;
    
    // Track API performance
    trackEvent(`api_${endpoint}_latency', {
      duration: Math.round(duration),
      status: response.status,
      success: response.ok
    });
    
    return response;
  } catch (error) {
    const duration = performance.now() - startTime;
    
    trackEvent(`api_${endpoint}_error'`, {
      duration: Math.round(duration),
      error: error.message
    });
    
    throw error;
  }
}

Feature Flags

Feature flags enable granular control over monetization features, allowing you to toggle features per user, percentage, or segment without deploying new code.

Local Feature Flags

For simple on/off features, use local storage with caching.

feature-flags.js - Local Implementation

const DEFAULT_FLAGS = {
  show_premium_upsell: true,
  new_pricing_page: false,
  annual_discount_banner: true,
  minimum_version_required: "1.0.0"
};

async function getFeatureFlags() {
  // Try cache first
  const cached = await chrome.storage.local.get(['feature_flags']);
  
  if (cached.feature_flags && cached.feature_flags.expires > Date.now()) {
    return cached.feature_flags.values;
  }
  
  // Fetch from server
  try {
    const response = await fetch(`https://api.yourdomain.com/v1/flags?uid=${userId}`);
    const serverFlags = await response.json();
    
    const mergedFlags = { ...DEFAULT_FLAGS, ...serverFlags };
    
    // Cache for 5 minutes
    await chrome.storage.local.set({
      feature_flags: {
        values: mergedFlags,
        expires: Date.now() + (5 * 60 * 1000)
      }
    });
    
    return mergedFlags;
  } catch (error) {
    return DEFAULT_FLAGS;
  }
}

async function isFeatureEnabled(flagName) {
  const flags = await getFeatureFlags();
  return flags[flagName] === true;
}

Server-Side License Validation

Robust license validation is critical for preventing piracy. Always validate licenses server-side and implement multiple layers of verification.

License Validation Flow

license-validation.js - Complete Flow

// Step 1: Generate license key client-side
function generateLicenseRequest(licenseKey) {
  const request = {
    license_key: licenseKey,
    extension_id: chrome.runtime.id,
    extension_version: chrome.runtime.getManifest().version,
    timestamp: Date.now(),
    machine_id: await getMachineId()
  };
  
  // Sign request to prevent tampering
  request.signature = signRequest(request);
  
  return request;
}

// Step 2: Validate on server
async function validateLicenseServer(request) {
  const response = await fetch('https://api.yourdomain.com/v1/license/validate', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json'
    },
    body: JSON.stringify(request)
  });
  
  if (!response.ok) {
    throw new Error(`License validation failed: ${response.status}`);
  }
  
  const result = await response.json();
  
  // Verify server response signature
  if (!verifyResponse(result)) {
    throw new Error('Invalid server response');
  }
  
  return result;
}

// Step 3: Cache with TTL
const LICENSE_CACHE_TTL = 3600; // 1 hour

async function getCachedLicenseStatus(forceRefresh = false) {
  const cached = await chrome.storage.local.get(['license_status']);
  
  if (!forceRefresh && cached.license_status) {
    const { status, expires } = cached.license_status;
    if (Date.now() < expires) {
      return status;
    }
  }
  
  // Refresh from server
  const licenseKey = await getStoredLicenseKey();
  const status = await validateLicenseServer(generateLicenseRequest(licenseKey));
  
  await chrome.storage.local.set({
    license_status: {
      status,
      expires: Date.now() + (LICENSE_CACHE_TTL * 1000)
    }
  });
  
  return status;
}

Webhook for License Events

Listen for license status changes via webhooks for real-time revocation.

webhook-handler.js - Server Endpoint

// Server-side webhook endpoint (Node.js example)
app.post('/webhooks/license', async (req, res) => {
  // Verify webhook signature
  const signature = req.headers['x-webhook-signature'];
  if (!verifyWebhookSignature(signature, req.body)) {
    return res.status(401).json({ error: 'Invalid signature' });
  }
  
  const { event_type, license_key, new_status } = req.body;
  
  switch (event_type) {
    case 'license_revoked':
    case 'license_expired':
    case 'license_refunded':
      // Push update to all users with this license
      await pushLicenseUpdate(license_key, { status: new_status });
      break;
      
    case 'license_upgraded':
      await pushLicenseUpdate(license_key, { 
        status: new_status,
        tier: req.body.new_tier 
      });
      break;
  }
  
  res.json({ received: true });
});

// Client-side: Listen for push updates
chrome.runtime.onMessage.addListener((message) => {
  if (message.type === 'license_update') {
    await chrome.storage.local.set({ license_status: message.status });
    
    // Notify user if license was revoked
    if (message.status.active === false) {
      showNotification('Your license has been revoked.');
    }
  }
});

Rate Limiting & Abuse Prevention

Protect your validation endpoints from abuse.

rate-limit.js - Client-Side Protection

// Implement client-side rate limiting
const RATE_LIMIT = {
  max_requests: 10,
  window_ms: 60000 // 1 minute
};

const requestHistory = new Map();

async function rateLimitedFetch(url, options) {
  const now = Date.now();
  const key = url;
  
  // Clean old requests
  const history = requestHistory.get(key) || [];
  const recentRequests = history.filter(t => now - t < RATE_LIMIT.window_ms);
  
  if (recentRequests.length >= RATE_LIMIT.max_requests) {
    throw new Error(`Rate limited. Try again in ${RATE_LIMIT.window_ms / 1000}s`);
  }
  
  recentRequests.push(now);
  requestHistory.set(key, recentRequests);
  
  return fetch(url, options);
}
